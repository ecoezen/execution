53template <::beman::execution::scope_token Token, ::beman::execution::sender Sender>
54struct associate_data {
55 using wrap_sender = ::std::remove_cvref_t<decltype(::std::declval<Token&>().wrap(::std::declval<Sender>()))>;
56 using assoc_t = decltype(::std::declval<Token&>().try_associate());
57 using sender_ref = ::std::unique_ptr<wrap_sender, decltype(std::ranges::destroy_at)>;
58
59 explicit associate_data(Token t, Sender&& s) : sender(t.wrap(::std::forward<Sender>(s))) {
60 sender_ref guard{::std::addressof(this->sender)};
61 this->assoc = t.try_associate();
62 if (this->assoc) {
63 static_cast<void>(guard.release());
64 }
65 }
66
67 explicit associate_data(::std::pair<assoc_t, sender_ref> parts) : assoc(::std::move(parts.first)) {
68 if (this->assoc) {
69 ::std::construct_at(::std::addressof(this->sender), ::std::move(*parts.second));
70 }
71 }
72
73 associate_data(const associate_data& other) noexcept(::std::is_nothrow_copy_constructible_v<wrap_sender> &&
74 noexcept(other.assoc.try_associate()))
75 requires ::std::copy_constructible<wrap_sender>
76 : assoc(other.assoc.try_associate()) {
77 if (this->assoc) {
78 ::std::construct_at(::std::addressof(this->sender), other.sender);
79 }
80 }
81
82 associate_data(associate_data&& other) noexcept(::std::is_nothrow_move_constructible_v<wrap_sender>)
83 : associate_data(::std::move(other).release()) {}
84
85 auto operator=(const associate_data&) -> associate_data& = delete;
86
87 auto operator=(associate_data&&) -> associate_data& = delete;
88
89 ~associate_data() {
90 if (this->assoc) {
91 ::std::destroy_at(::std::addressof(this->sender));
92 }
93 }
94
95 auto release() && noexcept -> ::std::pair<assoc_t, sender_ref> {
96 return {::std::move(assoc), sender_ref{assoc ? ::std::addressof(this->sender) : nullptr}};
97 }
98
99 assoc_t assoc;
100 union {
101 wrap_sender sender;
102 };
103};

141 template <typename AssociateData, typename Receiver>
142 struct op_state {
143 using assoc_t = typename AssociateData::assoc_t;
144 using sender_ref_t = typename AssociateData::sender_ref;
145 using op_t = ::beman::execution::connect_result_t<typename sender_ref_t::element_type, Receiver>;
146
147 assoc_t assoc;
148 union {
149 Receiver* rcvr;
150 op_t op;
151 };
152
153 explicit op_state(::std::pair<assoc_t, sender_ref_t> parts, Receiver& r)
154 : assoc(::std::move(parts.first)) {
155 if (assoc) {
156 ::std::construct_at(::std::addressof(op),
157 ::beman::execution::connect(::std::move(*parts.second), ::std::move(r)));
158 } else {
159 rcvr = ::std::addressof(r);
160 }
161 }
162
163 explicit op_state(AssociateData&& ad, Receiver& r) : op_state(::std::move(ad).release(), r) {}
164
165 explicit op_state(const AssociateData& ad, Receiver& r)
166 requires ::std::copy_constructible<AssociateData>
167 : op_state(AssociateData(ad).release(), r) {}
168
169 op_state(const op_state&) = delete;
170
171 op_state(op_state&&) = delete;
172
173 ~op_state() {
174 if (this->assoc) {
175 op.~op_t();
176 }
177 }
178
179 auto operator=(const op_state&) -> op_state& = delete;
180
181 auto operator=(op_state&&) -> op_state& = delete;
182
183 auto run() noexcept -> void {
184 if (this->assoc) {
185 ::beman::execution::start(this->op);
186 } else {
187 ::beman::execution::set_stopped(::std::move(*this->rcvr));
188 }
189 }
190 };

192 struct get_state_impl {
193 template <typename Sender, typename Receiver>
194 auto operator()(Sender&& sender, Receiver& receiver) const noexcept(
195 (::std::same_as<Sender, ::std::remove_cvref_t<Sender>> ||
196 ::std::is_nothrow_constructible_v<::std::remove_cvref_t<Sender>, Sender>) &&
197 execution::detail::nothrow_callable<::beman::execution::connect_t,
198 typename get_wrap_sender<::std::remove_cvref_t<Sender>>::type,
199 Receiver>) {
200 return op_state{
201 ::beman::execution::detail::forward_like<Sender>(::std::forward<Sender>(sender).template get<1>()),
202 receiver};
203 }
204 };
205 static constexpr auto get_state{get_state_impl{}};
206 struct start_impl {
207 auto operator()(auto& state, auto&&) const noexcept -> void { state.run(); }
208 };
209 static constexpr auto start{start_impl{}};
210 };
