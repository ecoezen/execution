58 auto try_push(Item* item) noexcept -> std::optional<Item*> {
59 void* ptr = head_.load();
60 if (ptr == this) {
61 return std::nullopt;
62 }
63 item->*Next = static_cast<Item*>(ptr);
64 while (!head_.compare_exchange_weak(ptr, item)) {
65 if (ptr == this) {
66 return std::nullopt;
67 }
68 item->*Next = static_cast<Item*>(ptr);
69 }
70 return static_cast<Item*>(ptr);
71 }
72
74 auto empty_and_not_shutdown() const noexcept -> bool { return head_.load() == nullptr; }

79 auto pop_all_and_shutdown() noexcept -> ::beman::execution::detail::intrusive_stack<Next> {
80 auto stack = ::beman::execution::detail::intrusive_stack<Next>{};
81 void* ptr = head_.exchange(this);
82 if (ptr == this) {
83 return stack;
84 }
85 auto item = static_cast<Item*>(ptr);
86 stack.head_ = item;
87 return stack;
88 }
