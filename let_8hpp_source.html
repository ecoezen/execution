100 template <::beman::execution::detail::movable_value Fun>
101 auto operator()(Fun&& fun) const {
102 return ::beman::execution::detail::make_sender_adaptor(*this, ::std::forward<Fun>(fun));
103 }
104 template <::beman::execution::sender Sender, ::beman::execution::detail::movable_value Fun>
105 auto operator()(Sender&& sender, Fun&& fun) const {
106 auto domain(::beman::execution::detail::get_domain_early(sender));
107 return ::beman::execution::transform_sender(
108 domain,
109 ::beman::execution::detail::make_sender(*this, ::std::forward<Fun>(fun), std::forward<Sender>(sender)));
110 }

112 template <typename Sender>
113 static auto env(Sender&& sender) {
114 if constexpr (requires {
115 ::beman::execution::detail::sched_env(
116 ::beman::execution::get_completion_scheduler<Completion>(
117 ::beman::execution::get_env(sender)));
118 })
119 return ::beman::execution::detail::sched_env(
120 ::beman::execution::get_completion_scheduler<Completion>(::beman::execution::get_env(sender)));
121 else if constexpr (requires {
122 ::beman::execution::detail::make_env(
123 ::beman::execution::get_domain,
124 ::beman::execution::get_domain(::beman::execution::get_env(sender)));
125 })
126 return ::beman::execution::detail::make_env(
127 ::beman::execution::get_domain, ::beman::execution::get_domain(::beman::execution::get_env(sender)));
128 else
129 return ::beman::execution::env<>{};
130 }
131 template <typename Sender, typename Env>
132 static auto join_env(Sender&& sender, Env&& e) -> decltype(auto) {
133 return ::beman::execution::detail::join_env(env(sender), ::beman::execution::detail::fwd_env(e));
134 }

158 template <typename>
159 struct get_completions;
160 template <template <typename...> class L, typename... C>
161 struct get_completions<L<C...>> {
162 using type = ::beman::execution::detail::meta::unique<
163 ::beman::execution::detail::meta::combine<decltype(::beman::execution::get_completion_signatures(
164 std::declval<typename apply_decayed<C>::sender_type>(), std::declval<Env>()))...>>;
165 };

167 using upstream_env = decltype(::beman::execution::detail::let_t<Comp>::join_env(::std::declval<Child>(),
168 ::std::declval<Env>()));

170 ::std::declval<Child>(), ::std::declval<upstream_env>()));
171 using other_completions = ::beman::execution::detail::meta::filter<other_completion, upstream_completions>;
172 using matching_completions =
173 ::beman::execution::detail::meta::filter<matching_completion, upstream_completions>;
174 using type = ::beman::execution::detail::meta::combine<typename get_completions<matching_completions>::type,
175 other_completions>;

228 static constexpr auto get_state{[]<typename Sender, typename Receiver>(Sender&& sender, Receiver&& receiver) {
229 auto& fun{sender.template get<1>()};
230 auto& child{sender.template get<2>()};
231
232 using fun_t = ::std::remove_cvref_t<decltype(fun)>;
233 using child_t = ::std::remove_cvref_t<decltype(child)>;
234 using env_t = decltype(::beman::execution::detail::let_t<Completion>::env(child));
235 using sigs_t =
236 ::beman::execution::completion_signatures_of_t<child_t, ::beman::execution::env_of_t<Receiver>>;
237 using comp_sigs_t = ::beman::execution::detail::meta::filter<filter_pred, sigs_t>;
238 using type_list_t = ::beman::execution::detail::meta::to<::std::variant, comp_sigs_t>;
239 using tuples_t = ::beman::execution::detail::meta::transform<to_tuple_t, type_list_t>;
240 using unique_t = ::beman::execution::detail::meta::unique<tuples_t>;
241 using args_t = ::beman::execution::detail::meta::prepend<std::monostate, unique_t>;
242 using ops_t = ::beman::execution::detail::meta::prepend<
243 ::std::monostate,
244 ::beman::execution::detail::meta::unique<::beman::execution::detail::meta::transform<
245 to_state<fun_t, ::std::remove_cvref_t<Receiver>, env_t>::template trans,
246 tuples_t>>>;

250 env_t env;
251 args_t args;
252 ops_t ops2;
253 };
254 return state_t{beman::execution::detail::allocator_aware_move(
255 ::beman::execution::detail::forward_like<Sender>(fun), receiver),
256 ::beman::execution::detail::let_t<Completion>::env(child),
257 {},
258 {}};

260 template <typename Receiver, typename... Args>
261 static auto
262 let_bind(auto& state, Receiver& receiver, Args&&... args) noexcept(noexcept(::beman::execution::connect(
263 ::std::apply(::std::move(state.fun),
264 ::std::move(state.args.template emplace<::beman::execution::detail::decayed_tuple<Args...>>(
265 ::std::forward<Args>(args)...))),
266 let_receiver<Receiver, decltype(state.env)>{receiver, state.env}))) {
267 using args_t = ::beman::execution::detail::decayed_tuple<Args...>;
268 auto mkop{[&] {
269 return ::beman::execution::connect(
270 ::std::apply(::std::move(state.fun),
271 ::std::move(state.args.template emplace<args_t>(::std::forward<Args>(args)...))),
272 let_receiver<Receiver, decltype(state.env)>{receiver, state.env});
273 }};
274 ::beman::execution::start(
275 state.ops2.template emplace<decltype(mkop())>(beman::execution::detail::emplace_from{mkop}));
276 }
277 struct complete_impl {
278 template <class Tag, class... Args>
279 auto operator()(auto, auto& state, auto& receiver, Tag, Args&&... args) const {
280 if constexpr (::std::same_as<Tag, Completion>) {
281 try {
282 let_bind(state, receiver, ::std::forward<Args>(args)...);
283 } catch (...) {
284 if constexpr (not noexcept(let_bind(state, receiver, ::std::forward<Args>(args)...)))
285 ::beman::execution::set_error(::std::move(receiver), ::std::current_exception());
286 }
287 } else {
288 Tag()(::std::move(receiver), ::std::forward<Args>(args)...);
289 }
290 }
291 };
292 static constexpr auto complete{complete_impl{}};
